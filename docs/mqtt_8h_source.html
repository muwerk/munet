137  void subsMsg(String topic, String msg, String originator) {
138  if (originator == "mqtt")
139  return; // avoid loops
140  if (mqttConnected) {
141  unsigned int len = msg.length() + 1;
142  String tpc = clientName + "/" + topic;
143  if (mqttClient.publish(tpc.c_str(), msg.c_str(), len)) {
144 #ifdef USE_SERIAL_DBG
145  Serial.println(
146  ("MQTT publish: " + topic + " | " + String(msg)).c_str());
147 #endif
148  } else {
149 #ifdef USE_SERIAL_DBG
150  Serial.println(("MQTT ERROR len=" + String(len) +
151  ", not published: " + topic + " | " +
152  String(msg))
153  .c_str());
154 #endif
155  if (len > 128) {
156 #ifdef USE_SERIAL_DBG
157  Serial.println("FATAL ERROR: you need to re-compile the "
158  "PubSubClient library and increase #define "
159  "MQTT_MAX_PACKET_SIZE.");
160 #endif
161  }
162  }
163  } else {
164 #ifdef USE_SERIAL_DBG
165  Serial.println(("MQTT can't publish, MQTT down: " + topic).c_str());
166 #endif
167  }
168  DynamicJsonBuffer jsonBuffer(200);
169  JsonObject &root = jsonBuffer.parseObject(msg);
170  if (!root.success()) {
171 #ifdef USE_SERIAL_DBG
172  Serial.println(
173  ("mqtt: Invalid JSON received: " + String(msg)).c_str());
174 #endif
175  return;
176  }
177  if (topic == "net/services/mqttserver") {
178  if (!bMqInit) {
179  mqttServer = root["server"].as<char *>();
180  bCheckConnection = true;
181  mqttClient.setServer(mqttServer.c_str(), 1883);
182  // give a c++11 lambda as callback for incoming mqtt
183  // messages:
184  std::function<void(char *, unsigned char *, unsigned int)> f =
185  [=](char *t, unsigned char *m, unsigned int l) {
186  this->mqttReceive(t, m, l);
187  };
188  // If this breaks for ESP32, patch pubsubclient, l.76:
189  // #if defined(ESP8266) or defined(ESP32)
190  // to allow functionals for callback signature
191  mqttClient.setCallback(f);
192  bMqInit = true;
193  }
194  }
195  if (topic == "net/network") {
196  String state = root["state"];
197  if (state == "connected") {
198  if (!netUp) {
199  netUp = true;
200  bCheckConnection = true;
201  }
202  } else {
203  netUp = false;
204  }
205  }
206  };

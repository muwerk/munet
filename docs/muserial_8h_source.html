219  void sendOut(String topic, String msg) {
220  T_HEADER th;
221  T_FOOTER tf;
222  unsigned char ccrc;
223  unsigned char startbyte = SOH;
224  unsigned char nul = 0x0;
225  memset(&th, 0, sizeof(th));
226  memset(&tf, 0, sizeof(tf));
227 
228  th.ver = VER;
229  th.num = blockNum++;
230  th.cmd = LinkCmd::MQTT;
231  unsigned int len = topic.length() + msg.length() + 2;
232  th.hLen = len / 256;
233  th.lLen = len % 256;
234  th.stx = STX;
235 
236  tf.etx = ETX;
237  tf.etx = EOT;
238 
239  ccrc = crc((const unsigned char *)&th, 6);
240  ccrc = crc((const unsigned char *)topic.c_str(), topic.length(), ccrc);
241  ccrc = crc(&nul, 1, ccrc);
242  ccrc = crc((const unsigned char *)msg.c_str(), msg.length(), ccrc);
243  ccrc = crc(&nul, 1, ccrc);
244  ccrc = crc((const unsigned char *)&fo, 1, ccrc);
245 
246  fo.crc = ccrc;
247 
248  pSerial->write(&startbyte, 1);
249  pSerial->write((unsigned char *)&th, 6);
250  pSerial->write((unsigned char *)topic.c_str(), topic.length());
251  pSerial->write(&nul, 1);
252  pSerial->write((unsigned char *)msg.c_str(), msg.length());
253  pSerial->write(&nul, 1);
254  pSerial->write((unsigned char *)&tf, 3);

287  void loop() {
288  unsigned char ccrc;
289  unsigned char c;
290  if (bCheckLink) {
291  if (time(nullptr) - lastPingSent > pingPeriod) {
292  ping();
293  }
294  while (pSerial->available() > 0) {
295  c = pSerial->read();
296  lastRead = time(nullptr);
297  switch (state) {
298  case SYNC:
299  if (c == SOH) {
300  state = HEADER;
301  pHd = (unsigned char *)&hd;
302  hLen = 0;
303  }
304  break;
305  case HEADER:
306  pSched->publish("muserial/link", "SOH");
307  pHd[hLen] = c;
308  hLen++;
309  if (hLen == 6) {
310  // XXX: check block number
311  if (hd.ver != VER || hd.stx != STX) {
312  state = SYNC;
313  } else {
314  msgLen = 256 * hd.hLen + hd.lLen;
315  msgBuf = (unsigned char *)malloc(msgLen);
316  curMsg = 0;
317  if (msgBuf) {
318  state = MSG;
319  allocated = true;
320  } else {
321  state = SYNC;
322  }
323  }
324  }
325  break;
326  case MSG:
327  pSched->publish("muserial/link", "MSG");
328  msgBuf[curMsg] = c;
329  ++curMsg;
330  if (curMsg == msgLen) {
331  state = CRC;
332  cLen = 0;
333  }
334  break;
335  case CRC:
336  pSched->publish("muserial/link", "CRC");
337  pFo[cLen] = c;
338  ++cLen;
339  if (cLen == 3) {
340  if (fo.etx != ETX || fo.eot != EOT) {
341  if (allocated && msgBuf != nullptr) {
342  free(msgBuf);
343  allocated = false;
344  }
345  state = SYNC;
346  continue;
347  }
348  }
349  ccrc = crc((const unsigned char *)&hd, 6);
350  ccrc = crc(msgBuf, msgLen, ccrc);
351  ccrc = crc((const unsigned char *)&fo, 1, ccrc);
352  if (ccrc != fo.crc) {
353  if (allocated && msgBuf != nullptr) {
354  free(msgBuf);
355  allocated = 0;
356  }
357  state = SYNC;
358  continue;
359  } else {
360  switch ((LinkCmd)hd.cmd) {
361  case MUPING:
362  if (strlen((const char *)&msgBuf[4]) < 10) {
363  remoteName = (const char *)&msgBuf[4];
364  lastMsg = time(nullptr);
365  if (!linkConnected) {
366  linkConnected = true;
367  pSched->publish(name + "/link", "connected");
368  }
369  } else {
370  if (allocated && msgBuf != nullptr) {
371  free(msgBuf);
372  allocated = 0;
373  }
374  state = SYNC;
375  pSched->publish(name + "/ping", remoteName);
376  continue;
377  }
378  break;
379  case MQTT:
380  if (strlen((const char *)msgBuf) + 2 < msgLen) {
381  const char *pM =
382  (const char *)&msgBuf[strlen((const char *)msgBuf) + 1];
383  if (strlen(pM) + strlen((const char *)msgBuf) + 2 < msgLen) {
384  internalPub((const char *)msgBuf, pM);
385  lastMsg = time(nullptr);
386  }
387  }
388  break;
389  }
390  if (allocated && msgBuf != nullptr) {
391  free(msgBuf);
392  allocated = 0;
393  }
394  state = SYNC;
395  }
396  break;
397  }
398  }
399  } else {
400  if (state != SYNC) {
401  if ((unsigned long)(time(nullptr) - lastRead) > readTimeout) {
402  state = SYNC;
403  if (linkConnected) {
404  pSched->publish(name + "/link", "disconnected");
405  }
406  linkConnected = false;
407  if (allocated) {
408  free(msgBuf);
409  msgBuf = nullptr;
410  }
411  }
412  } else {
413  if ((unsigned long)(time(nullptr) - lastMsg) > pingReceiveTimeout) {
414  if (linkConnected) {
415  pSched->publish(name + "/link", "disconnected");
416  }
417  linkConnected = false;
418  }
419  }
420  }
421  }

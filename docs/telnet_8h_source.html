41  void loop() {
42  connected = client.connected();
43  if (connected) {
44  int icount = client.available();
45  while (icount > 0) {
46  int iread = min(icount, (int)(sizeof(buffer) / sizeof(char)) - 1);
47  int idone = client.read((uint8_t *)buffer, iread);
48  if (idone) {
49  const char *pStart = buffer;
50  // our buffer is ALWAYS zero terminated!
51  buffer[idone] = 0;
52  for (char *pPtr = buffer; *pPtr; pPtr++) {
53  switch (*pPtr) {
54  case 9: // tab
55  // treat like space
56  *pPtr = 32;
57  break;
58  case 10: // line feed
59  case 13: // enter
60  // execute the command
61  *pPtr = 0;
62  DBGF("Executing %s\r\n", pStart);
63  args += pStart;
64  pStart = pPtr + 1;
65  DBGF("Rest is now at pos:%i v:%i s:%s\r\n", (int)(pStart - buffer),
66  (int)*pStart, pStart);
67  execute();
68  prompt();
69  break;
70  }
71  }
72  if (*pStart) {
73  DBGF("Appending rest at pos:%i v:%i s:%s\r\n", (int)(pStart - buffer),
74  (int)*pStart, pStart);
75  args += pStart;
76  }
77  icount -= idone;
78  }
79  }
80  } else {
81  if (!finished) {
82  DBGF("\rTelnet client %s:%u disconnected\r\n", client.remoteIP().toString().c_str(),
83  (unsigned int)client.remotePort());
84  client.stop();
85  finished = true;
86  }
87  }
88  }

112  void begin(Scheduler *_pSched) {
113  pSched = _pSched;
114  tID = pSched->add([this]() { this->loop(); }, "telnet", 60000); // 60ms
115  pSched->subscribe(tID, "net/network", [this](String topic, String msg, String originator) {
116  JSONVar jm = JSON.parse(msg);
117  if (JSON.typeof(jm) == "object" && JSON.typeof(jm["state"]) == "string") {
118  bool newconnected = !strcmp(jm["state"], "connected");
119  if (connected != newconnected) {
120  connected = newconnected;
121  if (connected) {
122  this->server.begin();
123  } else {
124  this->server.end();
125  }
126  }
127  }
128  });
129  }
130 
131  void loop() {
132  // cleanup finished consoles
133  for (unsigned int i = 0; i < consoles.length(); i++) {
134  TelnetConsole *pCon = consoles[i];
135  if (pCon->finished) {
136  DBGF("\rCleaning up console %i\r\n", i);
137  pCon->end();
138  delete pCon;
139  consoles.erase(i);
140  --i;
141  continue;
142  }
143  }
144  // accept connections
145  if (connected) {
146  for (WiFiClient client = server.available(); client; client = server.available()) {
147  DBGF("\rNew telnet connection from %s:%u\r\n", client.remoteIP().toString().c_str(),
148  client.remotePort());
149  TelnetConsole *pCon = new TelnetConsole(client);
150  consoles.add(pCon);
151  pCon->begin(pSched);
152  }
153  }
154  }

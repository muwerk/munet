441  void loop() {
442  switch (state) {
443  case NOTCONFIGURED:
444  if (timeDiff(tick10sec, millis()) > 10000) {
445  tick10sec = millis();
446  publishNetworks();
447  }
448  break;
449  case CONNECTINGAP:
450  if (WiFi.status() == WL_CONNECTED) {
451 #ifdef USE_SERIAL_DBG
452  Serial.println("Connected!");
453 #endif
454  state = CONNECTED;
455  IPAddress ip = WiFi.localIP();
456  ipAddress = String(ip[0]) + '.' + String(ip[1]) + '.' +
457  String(ip[2]) + '.' + String(ip[3]);
458  } else {
459  if (ustd::timeDiff(conTime, millis()) > conTimeout) {
460 #ifdef USE_SERIAL_DBG
461  Serial.println("Timeout connecting!");
462 #endif
463  if (bOnceConnected) {
464  --deathCounter;
465  if (deathCounter == 0) {
466 #ifdef USE_SERIAL_DBG
467  Serial.println("Final failure, restarting...");
468 #endif
469  if (bReboot)
470  ESP.restart();
471  }
472 #ifdef USE_SERIAL_DBG
473  Serial.println("reconnecting...");
474 #endif
475  WiFi.reconnect();
476  conTime = millis();
477  } else {
478 #ifdef USE_SERIAL_DBG
479  Serial.println("retrying to connect...");
480 #endif
481  if (initialCounter > 0) {
482  --initialCounter;
483  WiFi.reconnect();
484  conTime = millis();
485  state = CONNECTINGAP;
486 
487  } else {
488 #ifdef USE_SERIAL_DBG
489  Serial.println("Final connect failure, "
490  "configuration invalid?");
491 #endif
492  state = NOTCONFIGURED;
493  if (bReboot)
494  ESP.restart();
495  }
496  }
497  }
498  }
499  break;
500  case CONNECTED:
501  bOnceConnected = true;
502  deathCounter = RECONNECT_MAX_TRIES;
503  if (timeDiff(tick1sec, millis()) > 1000) {
504  tick1sec = millis();
505  if (WiFi.status() == WL_CONNECTED) {
506  long rssi = WiFi.RSSI();
507  if (rssival.filter(&rssi)) {
508  pSched->publish("net/rssi",
509  "{\"rssi\":" + String(rssi) + "}");
510  }
511  } else {
512  WiFi.reconnect();
513  state = CONNECTINGAP;
514  conTime = millis();
515  }
516  }
517  break;
518  default:
519  break;
520  }
521  if (state != oldState) {
522 #ifdef USE_SERIAL_DBG
523  char msg[128];
524  sprintf(msg, "Netstate: %d->%d", oldState, state);
525  Serial.println(msg);
526 #endif
527  if (state == NOTCONFIGURED || state == CONNECTED) {
528  if (signalLed != 0xff) {
529  digitalWrite(signalLed, HIGH); // Turn the LED off
530  }
531  } else {
532  if (signalLed != 0xff) {
533  digitalWrite(signalLed, LOW); // Turn the LED on
534  }
535  }
536  oldState = state;
537  publishNetwork();
538  }
539  }
